131 auto format(const AstarteDeviceSdk::AstarteMessage& msg, FormatContext& ctx) const {
132 auto out = ctx.out();
133
134 out = astarte_fmt::format_to(out, "{{interface: {}, path: {}", msg.get_interface(),
135 msg.get_path());
136
137 // check if the payload is an unset property, which is the only "empty" case
138 bool is_unset_prop = false;
139 const auto* prop =
140 std::get_if<AstarteDeviceSdk::AstartePropertyIndividual>(&msg.get_raw_data());
141 if (prop && !prop->get_value().has_value()) {
142 is_unset_prop = true;
143 }
144
145 if (!is_unset_prop) {
146 out = astarte_fmt::format_to(out, ", value: ");
147 std::visit([&out](const auto& arg) { out = astarte_fmt::format_to(out, "{}", arg); },
148 msg.get_raw_data());
149 }
150
151 return astarte_fmt::format_to(out, "}}");
152 }
153};

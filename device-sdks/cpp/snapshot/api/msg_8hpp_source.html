128 template <typename FormatContext>
129 auto format(const astarte::device::Message& msg, FormatContext& ctx) const {
130 auto out = ctx.out();
131
132 out = astarte_fmt::format_to(out, "{{interface: {}, path: {}", msg.get_interface(),
133 msg.get_path());
134
135 // check if the payload is an unset property, which is the only "empty" case
136 bool is_unset_prop = false;
137 const auto* prop = std::get_if<astarte::device::PropertyIndividual>(&msg.get_raw_data());
138 if (prop && !prop->get_value().has_value()) {
139 is_unset_prop = true;
140 }
141
142 if (!is_unset_prop) {
143 out = astarte_fmt::format_to(out, ", value: ");
144 std::visit([&out](const auto& arg) { out = astarte_fmt::format_to(out, "{}", arg); },
145 msg.get_raw_data());
146 }
147
148 return astarte_fmt::format_to(out, "}}");
149 }
150};

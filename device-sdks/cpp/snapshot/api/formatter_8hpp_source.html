52template <typename OutputIt>
53void format_base64(OutputIt& out, const std::vector<uint8_t>& data) {
54 static constexpr std::string_view base64_chars =
55 "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
56 "abcdefghijklmnopqrstuvwxyz"
57 "0123456789+/";
58
59 // Named constants to avoid magic numbers
60 static constexpr uint32_t mask_6bit = 0x3F;
61 static constexpr int shift_byte_0 = 16;
62 static constexpr int shift_byte_1 = 8;
63 static constexpr int shift_b64_0 = 18;
64 static constexpr int shift_b64_1 = 12;
65 static constexpr int shift_b64_2 = 6;
66
67 size_t idx = 0;
68 const size_t len = data.size();
69
70 out = astarte_fmt::format_to(out, "\"");
71
72 while (idx + 2 < len) {
73 const uint32_t chunk =
74 (data[idx] << shift_byte_0) | (data[idx + 1] << shift_byte_1) | data[idx + 2];
75 out = astarte_fmt::format_to(out, "{}{}{}{}", base64_chars[(chunk >> shift_b64_0) & mask_6bit],
76 base64_chars[(chunk >> shift_b64_1) & mask_6bit],
77 base64_chars[(chunk >> shift_b64_2) & mask_6bit],
78 base64_chars[chunk & mask_6bit]);
79 idx += 3;
80 }
81
82 if (idx < len) {
83 uint32_t chunk = data[idx] << shift_byte_0;
84 if (idx + 1 < len) {
85 chunk |= data[idx + 1] << shift_byte_1;
86 }
87
88 out = astarte_fmt::format_to(out, "{}{}", base64_chars[(chunk >> shift_b64_0) & mask_6bit],
89 base64_chars[(chunk >> shift_b64_1) & mask_6bit]);
90 if (idx + 1 < len) {
91 out = astarte_fmt::format_to(out, "{}=", base64_chars[(chunk >> shift_b64_2) & mask_6bit]);
92 } else {
93 out = astarte_fmt::format_to(out, "==");
94 }
95 }
96
97 out = astarte_fmt::format_to(out, "\"");
98}

106template <typename OutputIt>
107void format_timestamp(OutputIt& out, const std::chrono::system_clock::time_point& data) {
108 out = astarte_fmt::format_to(out, "\"");
109#if (__cplusplus >= 202002L) && (__has_include(<format>))
110 out = astarte_fmt::format_to(
111 out, "{}",
112 astarte_fmt::format("{0:%F}T{0:%T}Z",
113 std::chrono::time_point_cast<std::chrono::milliseconds>(data)));
114#else // (__cplusplus >= 202002L) && (__has_include(<format>))
115 const std::time_t time = std::chrono::system_clock::to_time_t(data);
116 const std::tm utc_tm = *std::gmtime(&time);
117 std::stringstream stream;
118 stream << std::put_time(&utc_tm, "%FT%T.000Z");
119 out = astarte_fmt::format_to(out, "{}", stream.str());
120#endif // (__cplusplus >= 202002L) && (__has_include(<format>))
121 out = astarte_fmt::format_to(out, "\"");
122}

131template <typename OutputIt, typename T>
132void format_data(OutputIt& out, const T& data) {
133 if constexpr (std::is_same_v<T, bool>) {
134 astarte_fmt::format_to(out, "{}", (data ? "true" : "false"));
135 } else if constexpr (std::is_same_v<T, std::string>) {
136 astarte_fmt::format_to(out, R"("{}")", data);
137 } else if constexpr (std::is_same_v<T, std::vector<uint8_t>>) {
138 format_base64(out, data);
139 } else if constexpr (std::is_same_v<T, std::chrono::system_clock::time_point>) {
140 format_timestamp(out, data);
141 } else { // default format case
142 astarte_fmt::format_to(out, "{}", data);
143 }
144}

153template <typename OutputIt, typename T>
154void format_vector(OutputIt& out, const std::vector<T>& data) {
155 out = astarte_fmt::format_to(out, "[");
156 for (size_t i = 0; i < data.size(); ++i) {
157 format_data(out, data[i]);
158 if (i != data.size() - 1) {
159 out = astarte_fmt::format_to(out, ", ");
160 }
161 }
162 out = astarte_fmt::format_to(out, "]");
163}

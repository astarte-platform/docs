29template <typename T>
30concept DataAllowedType = requires {
31 requires std::is_same_v<T, int32_t> || std::is_same_v<T, int64_t> || std::is_same_v<T, double> ||
32 std::is_same_v<T, bool> || std::is_same_v<T, std::string> ||
33 std::is_same_v<T, std::string_view> || std::is_same_v<T, std::vector<uint8_t>> ||

154 template <typename FormatContext>
155 // NOLINTNEXTLINE(readability-function-size)
156 auto format(const astarte::device::Data& data, FormatContext& ctx) const {
157 auto out = ctx.out();
158
159 if (std::holds_alternative<int32_t>(data.get_raw_data())) {
160 out = astarte_fmt::format_to(out, "{}", std::get<int32_t>(data.get_raw_data()));
161 } else if (std::holds_alternative<int64_t>(data.get_raw_data())) {
162 out = astarte_fmt::format_to(out, "{}", std::get<int64_t>(data.get_raw_data()));
163 } else if (std::holds_alternative<double>(data.get_raw_data())) {
164 out = astarte_fmt::format_to(out, "{}", std::get<double>(data.get_raw_data()));
165 } else if (std::holds_alternative<bool>(data.get_raw_data())) {
166 const auto* bool_str = (std::get<bool>(data.get_raw_data()) ? "true" : "false");
167 out = astarte_fmt::format_to(out, "{}", bool_str);
168 } else if (std::holds_alternative<std::string>(data.get_raw_data())) {
169 out = astarte_fmt::format_to(out, R"("{}")", std::get<std::string>(data.get_raw_data()));
170 } else if (std::holds_alternative<std::vector<uint8_t>>(data.get_raw_data())) {
171 astarte::device::utils::format_base64(out,
172 std::get<std::vector<uint8_t>>(data.get_raw_data()));
173 } else if (std::holds_alternative<std::chrono::system_clock::time_point>(data.get_raw_data())) {
174 astarte::device::utils::format_timestamp(
175 out, std::get<std::chrono::system_clock::time_point>(data.get_raw_data()));
176 } else if (std::holds_alternative<std::vector<int32_t>>(data.get_raw_data())) {
177 astarte::device::utils::format_vector(out,
178 std::get<std::vector<int32_t>>(data.get_raw_data()));
179 } else if (std::holds_alternative<std::vector<int64_t>>(data.get_raw_data())) {
180 astarte::device::utils::format_vector(out,
181 std::get<std::vector<int64_t>>(data.get_raw_data()));
182 } else if (std::holds_alternative<std::vector<double>>(data.get_raw_data())) {
183 astarte::device::utils::format_vector(out,
184 std::get<std::vector<double>>(data.get_raw_data()));
185 } else if (std::holds_alternative<std::vector<bool>>(data.get_raw_data())) {
186 astarte::device::utils::format_vector(out, std::get<std::vector<bool>>(data.get_raw_data()));
187 } else if (std::holds_alternative<std::vector<std::string>>(data.get_raw_data())) {
188 astarte::device::utils::format_vector(
189 out, std::get<std::vector<std::string>>(data.get_raw_data()));
190 } else if (std::holds_alternative<std::vector<std::vector<uint8_t>>>(data.get_raw_data())) {
191 astarte::device::utils::format_vector(

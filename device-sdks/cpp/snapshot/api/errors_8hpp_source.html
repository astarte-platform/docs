550template <>
551struct astarte_fmt::formatter<AstarteDeviceSdk::AstarteErrorBase> {
557 template <typename ParseContext>
558 constexpr auto parse(ParseContext& ctx) const {
559 return ctx.begin();
560 }
561
568 template <typename FormatContext>
569 auto format(const AstarteDeviceSdk::AstarteErrorBase& err, FormatContext& ctx) const {
570 auto out = astarte_fmt::format_to(ctx.out(), "{}: {}", err.type(), err.message());
571
572 std::string indent;
573 const AstarteDeviceSdk::AstarteErrorBase* current = &err;
574 while (const auto& nested = current->nested_error()) {
575 indent += " ";
576 out = astarte_fmt::format_to(out, "\n{}-> {}: {}", indent, nested->type(), nested->message());
577 current = nested.get();
578 }
579
580 return out;
581 }
